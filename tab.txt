SELECT TOP 1
name, count(name)
FROM orderss
JOIN promocodes ON orderss.promocode_id=promocodes.promocode_id
GROUP BY name
ORDER BY count(name) DESC

SELECT TOP 10
office_name, cookies
FROM buildings
JOIN consumption ON buildings.coffee_point_id=consumption.coffee_point_id
WHERE cookies<1000
ORDER BY cookies DESC
